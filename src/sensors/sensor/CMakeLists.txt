# Version du langage

# Export command for clangd

    src/sensor.cpp
    src/sensorCapture.cpp
    src/sensor.h
)

    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/src>
)

target_link_libraries(Sensor PUBLIC 
  Configuration
